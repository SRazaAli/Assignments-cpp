bubble sorting an array using pointers

void bulbulasort(){
	int arr[5] = {3,7,2,9,5};
	int *ptr_arr = arr;
	for(int i = 0;i<=5;i++){
		for(int j = 0;j<5;j++){
			if(*(ptr_arr+j)>*(ptr_arr+j+1)){
			 temp = *(ptr_arr + j);
			*(ptr_arr + j) = *(ptr_arr + j+1);
			  *(ptr_arr+j+1) = temp;
			
			}
		}
	}
	for(int i = 0;i<5;i++){
		cout << *(ptr_arr + i + 1)<<endl;
	}
}
